ttl救砖 ​

直接刷回官方固件 ​

sh
# 设置路由器ip
setenv ipaddr 192.168.1.1
# 设置tftp服务器地址
setenv serverip 192.168.1.8
# 从ttftp服务器下载分区文件
tftpboot mtd16 
# 刷写分区
flash rootfs
# 保存
saveenv

直接刷uboot ​

sh
# 只刷UBOOT
tftpboot uboot-cmiot-ax18.bin && flash 0:APPSBL
tftpboot uboot-cmiot-ax18.bin && flash 0:APPSBL_1
